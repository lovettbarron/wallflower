    **2. Add `list_jams_spatial()` to db/mod.rs:**
    Single SQL query using LEFT JOINs and GROUP_CONCAT to fetch all jams with their tempo, key, tags, collaborators, and instruments in one query. This avoids N+1 queries:
    ```sql
    SELECT j.id, j.filename, j.duration_seconds, j.imported_at, j.created_at,
           tr.bpm as tempo_bpm,
           kr.key_name, kr.scale as key_scale,
           GROUP_CONCAT(DISTINCT t.tag) as tags,
           GROUP_CONCAT(DISTINCT c.name) as collaborators,
           GROUP_CONCAT(DISTINCT i.name) as instruments
    FROM jams j
    LEFT JOIN tempo_results tr ON j.id = tr.jam_id
    LEFT JOIN key_results kr ON j.id = kr.jam_id
    LEFT JOIN jam_tags t ON j.id = t.jam_id
    LEFT JOIN jam_collaborators c ON j.id = c.jam_id
    LEFT JOIN jam_instruments i ON j.id = i.jam_id
    GROUP BY j.id
    ORDER BY j.imported_at DESC
    ```
    Parse GROUP_CONCAT results by splitting on comma. Handle NULL case (no tags/collabs/instruments) as empty vec.

    **2. Create `src/components/accessibility/HighContrastProvider.tsx`:**
    React context that detects `prefers-contrast: more` via `window.matchMedia`. Provides `useHighContrast()` hook returning `boolean`. This is the JS fallback for components that need programmatic contrast awareness (the CSS handles most cases). Include check for SSR (typeof window !== 'undefined'):
    ```tsx
    const HighContrastContext = createContext(false);

    export function HighContrastProvider({ children }: { children: ReactNode }) {
      const [isHighContrast, setIsHighContrast] = useState(false);
      useEffect(() => {
        const mq = window.matchMedia('(prefers-contrast: more)');
        setIsHighContrast(mq.matches);
        const handler = (e: MediaQueryListEvent) => setIsHighContrast(e.matches);
        mq.addEventListener('change', handler);
        return () => mq.removeEventListener('change', handler);
      }, []);
      return (
        <HighContrastContext.Provider value={isHighContrast}>
          {children}
        </HighContrastContext.Provider>
      );
    }

    export function useHighContrast(): boolean {
      return useContext(HighContrastContext);
    }
    ```

    **3. Create `src/components/accessibility/useRovingTabIndex.ts`:**
    Reusable hook per the research pattern. Takes `items: string[]` (list of IDs) and `activeId: string`. Returns `{ refs, handleKeyDown, getTabIndex }`. Supports ArrowDown/Up/Left/Right, Home, End. Calls `e.preventDefault()` only for arrow/Home/End keys (not Space or Enter to avoid blocking transport bar):
    ```tsx
    export function useRovingTabIndex<T extends HTMLElement>(
      items: string[],
      activeId: string,
      options?: { orientation?: 'horizontal' | 'vertical' | 'both' }
    )
    ```
    - `orientation: 'vertical'` uses ArrowDown/ArrowUp only
    - `orientation: 'horizontal'` uses ArrowLeft/ArrowRight only
    - `orientation: 'both'` (default) uses all four arrow keys
