Per D-02 and UI-SPEC "Section Markers on Waveform":

```tsx
"use client";
import type { SectionRecord } from "@/lib/types";

// Section type color palette from UI-SPEC
const SECTION_COLORS: Record<string, string> = {
  "Intro": "hsl(210 60% 65%)",    // Blue
  "Outro": "hsl(210 60% 65%)",    // Blue
  "Verse": "hsl(150 50% 55%)",    // Green (also Verse A, Verse B, etc.)
  "Chorus": "hsl(340 60% 65%)",   // Pink
  "Bridge": "hsl(45 70% 60%)",    // Gold
  "Loop": "hsl(270 50% 65%)",     // Purple
  "default": "hsl(220 10% 50%)",  // Muted -- unknown sections
};

function getSectionColor(label: string): string {
  for (const [key, color] of Object.entries(SECTION_COLORS)) {
    if (label.startsWith(key)) return color;
  }
  return SECTION_COLORS.default;
}

interface SectionMarkersProps {
  sections: SectionRecord[];
  totalDuration: number;     // Total audio duration in seconds
  containerWidth: number;    // Width of the waveform container in pixels
  showLabels?: boolean;      // True for WaveformDetail, false for WaveformOverview
}

export function SectionMarkers({ sections, totalDuration, containerWidth, showLabels = false }: SectionMarkersProps) {
  // Render section boundaries as absolutely positioned vertical lines
  // Line: 2px wide, full height, section color at 60% opacity
  // Label (if showLabels): positioned above waveform, 12px font, section color at full opacity
  //   Labels at start of each section
  //   If labels would overlap (less than 40px apart), omit later label (show on hover)
  // Hover on section region: background fills with section color at 8% opacity, tooltip shows "{label} ({duration})"
  // Click on section marker: emit event to seek playback to section start
}
```
