Add a `SearchFilter` struct and `search_jams` method:

```rust
#[derive(Debug, Clone, Default, Deserialize)]
#[serde(rename_all = "camelCase")]
pub struct SearchFilter {
    pub query: Option<String>,          // Free-text search (D-15)
    pub keys: Option<Vec<String>>,      // Multi-select key filter (D-14), e.g. ["Bb minor", "C major"]
    pub tempo_min: Option<f64>,         // BPM range lower bound (D-13)
    pub tempo_max: Option<f64>,         // BPM range upper bound (D-13)
    pub tags: Option<Vec<String>>,      // Tag filter
    pub collaborators: Option<Vec<String>>,
    pub instruments: Option<Vec<String>>,
    pub date_from: Option<String>,      // ISO date string
    pub date_to: Option<String>,        // ISO date string
    pub location: Option<String>,
}
```

Implement `pub fn search_jams(&self, filter: &SearchFilter) -> Result<Vec<JamRecord>>`:
- Build a dynamic SQL query with parameterized WHERE clauses
- Filters combine with AND logic (D-12)
- For `query` (free-text, D-15): use FTS5 `jam_search MATCH ?` to get matching jam_ids, then JOIN
- For `keys`: JOIN jam_key and filter WHERE `key_name || ' ' || scale IN (?, ?, ...)`
- For `tempo_min`/`tempo_max`: JOIN jam_tempo and filter WHERE `bpm BETWEEN ? AND ?`
- For `tags`: JOIN jam_tags and filter WHERE `tag IN (?, ?, ...)`
- For `collaborators`: JOIN jam_collaborators and filter WHERE `name IN (?, ?, ...)`
- For `instruments`: JOIN jam_instruments and filter WHERE `name IN (?, ?, ...)`
- For `date_from`/`date_to`: filter jams WHERE `imported_at BETWEEN ? AND ?`
- For `location`: filter WHERE `location LIKE ?`
- Order by imported_at DESC (most recent first)

Also add helper methods:
- `pub fn get_distinct_keys(&self) -> Result<Vec<String>>` -- SELECT DISTINCT key_name || ' ' || scale FROM jam_key
- `pub fn get_distinct_tags(&self) -> Result<Vec<String>>` -- SELECT DISTINCT tag FROM jam_tags
- `pub fn get_distinct_collaborators(&self) -> Result<Vec<String>>` -- SELECT DISTINCT name FROM jam_collaborators  
- `pub fn get_distinct_instruments(&self) -> Result<Vec<String>>` -- SELECT DISTINCT name FROM jam_instruments
- `pub fn get_tempo_range(&self) -> Result<(f64, f64)>` -- SELECT MIN(bpm), MAX(bpm) FROM jam_tempo

#[command]
pub fn search_jams(app: AppHandle, filter: SearchFilter) -> Result<Vec<JamRecord>, String> {
    let state = app.state::<AppState>();
    let db = state.db.lock().map_err(|e| e.to_string())?;
    db.search_jams(&filter).map_err(|e| e.to_string())
}

#[command]
pub fn get_filter_options(app: AppHandle) -> Result<serde_json::Value, String> {
    let state = app.state::<AppState>();
    let db = state.db.lock().map_err(|e| e.to_string())?;
    let keys = db.get_distinct_keys().map_err(|e| e.to_string())?;
    let tags = db.get_distinct_tags().map_err(|e| e.to_string())?;
    let collaborators = db.get_distinct_collaborators().map_err(|e| e.to_string())?;
    let instruments = db.get_distinct_instruments().map_err(|e| e.to_string())?;
    let tempo_range = db.get_tempo_range().unwrap_or((60.0, 200.0));
    Ok(serde_json::json!({
        "keys": keys,
        "tags": tags,
        "collaborators": collaborators,
        "instruments": instruments,
        "tempoMin": tempo_range.0,
        "tempoMax": tempo_range.1,
    }))
}

**Step 1: Create src/lib/stores/library.ts (or extend if exists).**
Zustand store managing filter state:

```typescript
import { create } from "zustand";
import type { SearchFilter } from "@/lib/types";

interface LibraryState {
  filter: SearchFilter;
  setFilter: (filter: Partial<SearchFilter>) => void;
  clearFilter: () => void;
  clearFilterField: (field: keyof SearchFilter) => void;
  hasActiveFilters: boolean;
}

export const useLibraryStore = create<LibraryState>((set, get) => ({
  filter: {},
  setFilter: (partial) => set((state) => ({
    filter: { ...state.filter, ...partial },
    hasActiveFilters: Object.values({ ...state.filter, ...partial }).some(
      (v) => v !== undefined && v !== null && (Array.isArray(v) ? v.length > 0 : v !== "")
    ),
  })),
  clearFilter: () => set({ filter: {}, hasActiveFilters: false }),
  clearFilterField: (field) => set((state) => {
    const newFilter = { ...state.filter };
    delete newFilter[field];
    return {
      filter: newFilter,
      hasActiveFilters: Object.values(newFilter).some(
        (v) => v !== undefined && v !== null && (Array.isArray(v) ? v.length > 0 : v !== "")
      ),
    };
  }),
  hasActiveFilters: false,
}));
```
